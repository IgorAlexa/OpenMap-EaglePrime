<script src="https://unpkg.com/leaflet@1.7.1/dist/leaflet.js"></script>

<script>

var currentFloor = 7;
var maxFloor = 15;
var minFloor = 0;

var map = L.map('map', {
crs: L.CRS.Simple,
minZoom:-2
});

var bounds = [[0,0],[2048,2048]];

var layer = L.imageOverlay("OpenMap/foxworld_" + currentFloor + ".png", bounds).addTo(map);

map.fitBounds(bounds);

function updateFloor() {

map.removeLayer(layer);

layer = L.imageOverlay("OpenMap/foxworld_" + currentFloor + ".png", bounds).addTo(map);

document.getElementById("floor").innerText = currentFloor;

}

function floorUp(){

if(currentFloor < maxFloor){
currentFloor++;
updateFloor();
}

}

function floorDown(){

if(currentFloor > minFloor){
currentFloor--;
updateFloor();
}

}

</script>
